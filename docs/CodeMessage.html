def <=>(other)
  f = @filename <=> other.filename
  l = @linenumber.to_i <=> other.linenumber.to_i
  c = @colnumber.to_i <=> other.colnumber.to_i
  mt = @messagetype <=> other.messagetype
  m = @message[0..10] <=> other.message[0..10]

  ret = if f != 0
          f
        elsif l != 0
          l
        elsif c != 0
          c
        elsif mt != 0
          mt
        else
          m
        end
  ret
end
